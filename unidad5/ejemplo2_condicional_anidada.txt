Inicio
    Real nota
    Cadena clasificacion

    // Condicionales anidadas
    Si (nota >= 18) Entonces
        clasificacion = "Excelente"
    Sino
        Si (nota >= 15) Entonces
            clasificacion = "Bueno"
        Sino
            Si (nota >= 10) Entonces
                clasificacion = "Aprobado"
            Sino
                Si (nota >= 5) Entonces
                    clasificacion = "Insuficiente"
                Sino
                    clasificacion = "Deficiente"
                Fin Si
            Fin Si
        Fin Si
    Fin Si

    Escribir "Nota: ", nota
