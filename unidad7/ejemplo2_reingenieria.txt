========================================
DESPUES DE REINGENIERIA (codigo mejorado):
========================================

Constante Real TASA_IVA = 0.16

Funcion Real calcularSubtotal(Real precio, Entero cantidad)
    Retornar precio * cantidad
Fin Funcion

Funcion Real calcularIVA(Real subtotal)
    Retornar subtotal * TASA_IVA
Fin Funcion

Inicio
    Real precioUnitario, subtotal, iva, total
    Entero cantidad

    Escribir "=== Factura de Venta ==="
    Escribir "Precio unitario del producto: "
    Leer precioUnitario
    Escribir "Cantidad de productos: "
    Leer cantidad

    subtotal = calcularSubtotal(precioUnitario, cantidad)
    iva = calcularIVA(subtotal)
    total = subtotal + iva

    Escribir "--- Detalle de Factura ---"
    Escribir "Subtotal: ", subtotal
    Escribir "IVA (16%): ", iva
    Escribir "Total a pagar: ", total
Fin
